Konfigurasi Integrasi Google Sheets

Hubungkan pangkalan data Google Sheet anda dengan aplikasi kuiz ini.

Sistem akan secara automatik menggunakan mod simulasi tempatan jika ruangan ini dikosongkan.

Panduan 5 Langkah Integrasi Google Sheets & Drive

  1. Buka Google Sheets baharu dalam akaun Google anda.
  2. Klik pada menu atas Extensions (Sambungan) > Apps Script.
  3. Padam semua kod asal di dalam editor, dan salin (copy) kod di bawah sepenuhnya.
  4. Klik ikon Deploy (Atur Sedia) di bahagian atas kanan > pilih New Deployment (Atur sedia baharu) > pilih jenis Web App (Aplikasi Web).
    • Setkan Execute as kepada Me (akaun Google anda).
    • Setkan Who has access kepada Anyone (Sesiapa sahaja).
  5. Tekan Deploy, luluskan kebenaran keselamatan, kemudian salin URL Web App yang diberikan dan tampalkan ke dalam kotak tetapan di atas.

Kod Google Apps Script

function doPost(e) {
  // Tetapan ID Google Sheet dan ID Google Drive secara statik
  var sheetId = "1ATGIIR9YadS9Qjr7NGJvPCTzx0oo3VRdbmXFj51oudU";
  var folderId = "1YTf1gt1dWxqPHJqPsWhMX2AhbyQhLFwk";
  
  var ss = SpreadsheetApp.openById(sheetId);
  var sheet = ss.getActiveSheet();
  
  // Auto buat Header jika tiada di Google Sheet
  if (sheet.getLastRow() === 0) {
    sheet.appendRow([
      "Tarikh & Masa", 
      "Nama Penuh", 
      "No. Kad Pengenalan", 
      "Institusi Pengajian", 
      "Skor", 
      "Peratus Ketepatan",
      "Tempoh Masa", 
      "ID Resit Unik", 
      "Ringkasan Jawapan",
      "Pautan Resit (Google Drive)"
    ]);
  }
  
  try {
    var params = JSON.parse(e.postData.contents);
    
    // Bina struktur teks resit rasmi menggunakan waktu tempatan dari peranti peserta
    var fileContent = "====================================\n" +
                      "      RESIT RASMI KEPUTUSAN KUIZ\n" +
                      "====================================\n" +
                      "ID Transaksi : #" + params.resitId + "\n" +
                      "Tarikh & Masa: " + params.timestamp + "\n\n" +
                      "MAKLUMAT PESERTA:\n" +
                      "Nama      : " + params.nama + "\n" +
                      "No. KP    : " + params.noKp + "\n" +
                      "Institusi : " + params.institusi + "\n\n" +
                      "BUTIRAN KEPUTUSAN:\n" +
                      "Skor Akhir: " + params.skor + "\n" +
                      "Peratus   : " + params.peratus + "%\n" +
                      "Masa      : " + params.tempoh + "\n\n" +
                      "RINGKASAN DETIL JAWAPAN:\n" +
                      params.ringkasanJawapan.replace(/,/g, "\n") + "\n" +
                      "====================================\n" +
                      "Janaan Sistem Automatik - E-Math Quiz";
                      
    // Simpan fail resit ke Google Drive folder yang ditetapkan
    var folder = DriveApp.getFolderById(folderId);
    var fileName = "Resit_" + params.nama.replace(/\s+/g, "_") + "_" + params.resitId + ".txt";
    var file = folder.createFile(fileName, fileContent, MimeType.PLAIN_TEXT);
    
    // Set kebenaran akses supaya boleh dilihat oleh admin
    file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    var fileUrl = file.getUrl();
    
    // Simpan rekod peserta ke baris baharu
    sheet.appendRow([
      params.timestamp, // Menggunakan waktu yang dihantar dari peranti peserta untuk ketepatan zon masa Malaysia
      params.nama,
      "'" + params.noKp, // Format no kp sebagai teks
      params.institusi,
      params.skor,
      params.peratus + "%",
      params.tempoh,
      params.resitId,
      params.ringkasanJawapan,
      fileUrl
    ]);
    
    return ContentService.createTextOutput(JSON.stringify({result: "success"}))
                         .setMimeType(ContentService.MimeType.JSON);
  } catch(error) {
    return ContentService.createTextOutput(JSON.stringify({result: "error", error: error.toString()}))
                         .setMimeType(ContentService.MimeType.JSON);
  }
}
